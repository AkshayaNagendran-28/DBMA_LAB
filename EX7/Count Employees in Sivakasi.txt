Count Employees in Sivakasi with Exception Handlin

DECLARE
    CURSOR emp_cursor IS
        SELECT e.empno, e.ename, e.salary, e.city, e.deptno
        FROM employee e
        WHERE UPPER(e.city) = 'SIVAKASI'
        AND e.salary > (SELECT AVG(salary) 
                        FROM employee e2 
                        WHERE e2.deptno = e.deptno);
    
    v_empno employee.empno%TYPE;
    v_ename employee.ename%TYPE;
    v_salary employee.salary%TYPE;
    v_city employee.city%TYPE;
    v_deptno employee.deptno%TYPE;
    v_count NUMBER := 0;
    
BEGIN
    DBMS_OUTPUT.PUT_LINE('Employees in Sivakasi earning above dept average:');
    DBMS_OUTPUT.PUT_LINE('---------------------------------------------------');
    
    OPEN emp_cursor;
    
    LOOP
        FETCH emp_cursor INTO v_empno, v_ename, v_salary, v_city, v_deptno;
        EXIT WHEN emp_cursor%NOTFOUND;
        
        v_count := v_count + 1;
        DBMS_OUTPUT.PUT_LINE('Emp No: ' || v_empno || 
                             ' | Name: ' || v_ename || 
                             ' | Salary: ' || v_salary ||
                             ' | Dept: ' || v_deptno);
    END LOOP;
    
    CLOSE emp_cursor;
    
    DBMS_OUTPUT.PUT_LINE('---------------------------------------------------');
    DBMS_OUTPUT.PUT_LINE('Total Count: ' || v_count);
    
    IF v_count = 0 THEN
        DBMS_OUTPUT.PUT_LINE('No employees found matching criteria');
    END IF;
    
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Error: No data found');
        IF emp_cursor%ISOPEN THEN
            CLOSE emp_cursor;
        END IF;
        
    WHEN TOO_MANY_ROWS THEN
        DBMS_OUTPUT.PUT_LINE('Error: Too many rows returned');
        IF emp_cursor%ISOPEN THEN
            CLOSE emp_cursor;
        END IF;
        
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Error Code: ' || SQLCODE);
        DBMS_OUTPUT.PUT_LINE('Error Message: ' || SQLERRM);
        IF emp_cursor%ISOPEN THEN
            CLOSE emp_cursor;
        END IF;
END;
