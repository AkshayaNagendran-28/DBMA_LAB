Cursor with Attributes
 (Name starts with 'a')



DECLARE
    CURSOR emp_cursor IS
        SELECT empno, ename, salary 
        FROM employee 
        WHERE ename LIKE 'A%' OR ename LIKE 'a%';
    
    v_empno employee.empno%TYPE;
    v_ename employee.ename%TYPE;
    v_salary employee.salary%TYPE;
BEGIN
    OPEN emp_cursor;
    
    DBMS_OUTPUT.PUT_LINE('Cursor is Open: ' || 
        CASE WHEN emp_cursor%ISOPEN THEN 'TRUE' ELSE 'FALSE' END);
    
    LOOP
        FETCH emp_cursor INTO v_empno, v_ename, v_salary;
        EXIT WHEN emp_cursor%NOTFOUND;
        
        DBMS_OUTPUT.PUT_LINE('Employee: ' || v_empno || ' | ' || 
                             v_ename || ' | ' || v_salary);
        DBMS_OUTPUT.PUT_LINE('Row Count: ' || emp_cursor%ROWCOUNT);
    END LOOP;
    
    DBMS_OUTPUT.PUT_LINE('Total Rows Fetched: ' || emp_cursor%ROWCOUNT);
    DBMS_OUTPUT.PUT_LINE('Not Found: ' || 
        CASE WHEN emp_cursor%NOTFOUND THEN 'TRUE' ELSE 'FALSE' END);
    
    CLOSE emp_cursor;
    
    DBMS_OUTPUT.PUT_LINE('Cursor is Open: ' || 
        CASE WHEN emp_cursor%ISOPEN THEN 'TRUE' ELSE 'FALSE' END);
END;
/
