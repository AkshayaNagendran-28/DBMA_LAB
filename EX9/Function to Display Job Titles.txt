Function to Display Job Titles

CREATE OR REPLACE FUNCTION func_display_job_titles
RETURN SYS_REFCURSOR
AS
    v_refcursor SYS_REFCURSOR;
    v_sql VARCHAR2(500);
BEGIN
    v_sql := 'SELECT DISTINCT job FROM employees ORDER BY job';
    OPEN v_refcursor FOR v_sql;
    RETURN v_refcursor;
END func_display_job_titles;
/

-- Execute
SET SERVEROUTPUT ON;
DECLARE
    v_cursor SYS_REFCURSOR;
    v_job employees.job%TYPE;
BEGIN
    DBMS_OUTPUT.PUT_LINE('====== JOB TITLES ======');
    v_cursor := func_display_job_titles;
    
    LOOP
        FETCH v_cursor INTO v_job;
        EXIT WHEN v_cursor%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE(v_job);
    END LOOP;
    
    CLOSE v_cursor;
END;
/
