Prevent Updates During Specific Hours

CREATE OR REPLACE TRIGGER trg_restrict_update_hours
BEFORE UPDATE OF salary ON employees
FOR EACH ROW
DECLARE
    v_hour NUMBER;
BEGIN
    v_hour := TO_NUMBER(TO_CHAR(SYSDATE, 'HH24'));
    
    -- Prevent updates between 6 PM (18) and 8 AM (8)
    IF v_hour >= 18 OR v_hour < 8 THEN
        RAISE_APPLICATION_ERROR(-20001, 
            'Updates not allowed between 6 PM and 8 AM');
    END IF;
END;
/
